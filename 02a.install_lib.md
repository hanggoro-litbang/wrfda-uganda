# Install Libraries (GNU)
## Download All Libraries
```console
cd ~
mv raw raw.old
mkdir raw
cd raw
export prefix=~/wrflibsgnu
```
edit .bashrc
```console
vi ~/.bashrc
```
add lines
```console
#export PATH=$PATH:/home/${USER}/wrflibs/bin
#export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/home/${USER}/wrflibs/lib:/home/${USER}/wrflibs/lib64
export PATH=$PATH:/home/${USER}/wrflibsgnu/bin
export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/home/${USER}/wrflibsgnu/lib:/home/${USER}/wrflibsgnu/lib64
```
```console
wget https://zlib.net/fossils/zlib-1.2.9.tar.gz
wget https://support.hdfgroup.org/ftp/lib-external/szip/2.1.1/src/szip-2.1.1.tar.gz
wget https://github.com/HDFGroup/hdf5/releases/download/hdf5_1.14.6/hdf5-1.14.6.tar.gz
wget https://github.com/Unidata/netcdf-c/archive/refs/tags/v4.7.4.tar.gz
wget https://github.com/Unidata/netcdf-fortran/archive/refs/tags/v4.5.3.tar.gz
wget https://sourceforge.net/projects/libpng/files/libpng16/older-releases/1.6.37/libpng-1.6.37.tar.gz
wget https://www.ece.uvic.ca/~frodo/jasper/software/jasper-2.0.14.tar.gz
```
## Or you can copy from path below
```console
cd ~/raw
cp /scratch/wido/uganda/rawlib/* .
```
# Install MPICH or openmpi (skip)
You may skip this step, since mpi provided via module environment
```console
cd ~
mkdir -p wrflibsgnu
cd ~/raw
```
```console
tar -zxvf openmpi-4.1.4.tar.gz
cd openmpi-4.1.4
./configure --prefix=$prefix
make
make install
``` 
or 
```console
./configure --prefix=${openmpi install dir}  --with-pmi={slurm install dir}
make
make install
```
save and exit

# Install Jasper
```console
cd ~/raw
LD_PRELOAD=/lib64/libm.so.6 tar -zxvf jasper-2.0.14.tar.gz
cd jasper-2.0.14
mkdir Build
cd Build
```
```console
cmake -H../ -B . -DCMAKE_INSTALL_PREFIX=$prefix JAS_ENABLE_SHARED
cmake --build .
cmake --build . --target install
```
